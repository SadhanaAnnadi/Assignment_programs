using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;
using System.IO;

namespace ASPWebapplication_Example
{
    public partial class DownloadFile : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {

        }

        protected void btn1_Click(object sender, EventArgs e)
        {
            var filepath = @"C:\Users\sadhana\Desktop\mydog.jpeg";
            FileInfo file = new FileInfo(filepath);
            if(file.Exists)
            {
                //clear the Response
                Response.Clear();
                //Add Header by specifying file name
                Response.AddHeader("content Dispositon", "attachment;filenamee" + file.Name);
                //add header for content length
                Response.AddHeader("content-Length", file.Length.ToString());
                //specify the content-type
                Response.ContentType = "jpeg/plain";
                //clearing flush
                Response.Flush();
                //transmit the file
                Response.TransmitFile(file.FullName);
                Response.End();
            }
            else
            {
                label1.Text = "Requested file is not available to download";
            }
        }
    }
}
